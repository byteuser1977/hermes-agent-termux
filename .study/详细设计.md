## 1. AIAgent 核心类设计

### 1.1 类结构与初始化

```python
class AIAgent:

        # ... provider, api_mode, callbacks, routing params
    )
```

**关键属性**：
- `model`: 当前使用的 LLM 模型标识符
- `max_iterations`: 最大 API 调用次数限制
- `enabled_toolsets` / `disabled_toolsets`: 工具集启用/禁用列表
- `platform`: 运行平台标识（CLI、Telegram 等）
- `session_id`: 会话 ID，用于持久化

### 1.2 核心方法

#### `chat(message: str) -> str`
简单对话接口，返回最终响应字符串。

**流程**：
1. 构建初始消息列表（system + user）
2. 调用 `run_conversation()`
3. 提取 `final_response` 并返回

#### `run_conversation(...) -> dict`
完整对话接口，返回包含 `final_response` 和 `messages` 的字典。

**循环实现**：
```python
def run_conversation(self, user_message, system_message=None, 
                     conversation_history=None, task_id=None):
    messages = self._build_initial_messages(...)
    api_call_count = 0
    
    while (api_call_count < self.max_iterations and 
           self.iteration_budget.remaining > 0):
        # 1. 调用 LLM
        response = client.chat.completions.create(
            model=self.model,
            messages=messages,
            tools=self.tool_schemas
        )
        
        # 2. 检查是否有工具调用
        if response.choices[0].message.tool_calls:
            for tool_call in response.choices[0].message.tool_calls:
                # 3. 执行工具
                result = handle_function_call(
                    tool_call.name, 
                    tool_call.arguments, 
                    task_id
                )
                # 4. 添加工具结果到消息历史
                messages.append({
                    "role": "tool",
                    "tool_call_id": tool_call.id,
                    "content": result
                })
            api_call_count += 1
        else:
            # 5. 无工具调用，返回最终响应
            return {
                "final_response": response.choices[0].message.content,
                "messages": messages
            }
    
    # 达到迭代上限
    return {
        "final_response": "已达到最大迭代次数",
        "messages": messages
    }
```

## 2. 工具系统设计

### 2.1 工具注册表 (`tools/registry.py`)

```python
class ToolRegistry:
    def __init__(self):
        self._tools = {}  # name -> {schema, handler, toolset}
    
    def register(self, name, toolset, schema, handler):
        """注册一个工具"""
        self._tools[name] = {
            "schema": schema,
            "handler": handler,
            "toolset": toolset
        }
    
    def get_tool(self, name):
        """获取工具定义"""
        return self._tools.get(name)
    
    def list_tools(self, toolsets=None):
        """列出指定工具集中的工具"""
        if toolsets is None:
            return list(self._tools.values())
        return [
            tool for name, tool in self._tools.items()
            if tool["toolset"] in toolsets
        ]

registry = ToolRegistry()  # 单例
```

### 2.2 工具发现 (`model_tools.py`)

```python
def _discover_tools(enabled_toolsets, disabled_toolsets):
    """
    从工具集中发现可用工具
    
    流程：
    1. 加载所有工具模块（导入 tools/*.py）
    2. 根据 enabled_toolsets 过滤
    3. 排除 disabled_toolsets
    4. 检查 API 密钥和需求
    5. 返回可用的工具 schema 列表
    """
    available_tools = []
    
    for tool_name, tool_info in registry._tools.items():
        toolset = tool_info["toolset"]
        
        # 过滤逻辑
        if enabled_toolsets and toolset not in enabled_toolsets:
            continue
        if disabled_toolsets and toolset in disabled_toolsets:
            continue
        
        # 检查依赖
        if not tool_info["handler"].check_requirements():
            continue
        
        available_tools.append(tool_info["schema"])
    
    return available_tools
```

### 2.3 工具分发 (`model_tools.py`)

```python
def handle_function_call(tool_name, tool_args, task_id=None):
    """
    根据工具名称分发到具体实现
    
    流程：
    1. 从 registry 获取工具 handler
    2. 解析参数
    3. 执行工具函数
    4. 返回 JSON 格式结果
    """
    tool_info = registry.get_tool(tool_name)
    if not tool_info:
        return json.dumps({"error": f"Unknown tool: {tool_name}"})
    
    handler = tool_info["handler"]
    args = json.loads(tool_args) if isinstance(tool_args, str) else tool_args
    
    try:
        result = handler.function(task_id=task_id, **args)
        return json.dumps({"success": True, "data": result})
    except Exception as e:
        return json.dumps({"success": False, "error": str(e)})
```

### 4.1 主循环 (`gateway/run.py`)

```python
class GatewayRunner:
    def __init__(self):
        self.config = load_gateway_config()
        self.session_store = SessionStore()
        self.platforms = self._init_platforms()
    
    async def run(self):
        """主循环"""
        await self._start_all_platforms()
        
        while True:
            # 等待事件
            event = await self.event_queue.get()
            await self._dispatch_event(event)
    
    async def _dispatch_event(self, event):
        """分发事件"""
        if event.type == "message":
            await self._handle_message(event)
        elif event.type == "command":
            await self._handle_command(event)
    
    async def _handle_message(self, event):
        """处理用户消息"""
        session = self.session_store.get_or_create(event.session_id)
        
        # 构建 agent
        agent = AIAgent(
            platform=event.platform,
            session_id=session.id,
            ...
        )
        
        # 运行对话
        response = await asyncio.to_thread(agent.chat, event.text)
        
        # 发送响应
        await self.platforms[event.platform].send_message(
            event.chat_id, response
        )
```

## 5. 状态管理系统 (`hermes_state.py`)

### 5.1 SessionDB 类

```python
class SessionDB:
    def __init__(self, db_path=None):
        self.db_path = db_path or get_hermes_home() / "sessions.db"
        self._init_db()
    
    def _init_db(self):
        """初始化 SQLite 数据库和 FTS5 索引"""
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS sessions (
                    id TEXT PRIMARY KEY,
                    platform TEXT,
                    created_at INTEGER,
                    updated_at INTEGER,
                    metadata TEXT
                )
            """)
            
            conn.execute("""
                CREATE TABLE IF NOT EXISTS messages (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    session_id TEXT,
                    role TEXT,
                    content TEXT,
                    timestamp INTEGER,
                    FOREIGN KEY (session_id) REFERENCES sessions(id)
                )
            """)
            
            # FTS5 全文搜索
            conn.execute("""
                CREATE VIRTUAL TABLE IF NOT EXISTS messages_fts 
                USING fts5(content, content_rowid=id)
            """)
    
    def create_session(self, platform, session_id=None):
        """创建新会话"""
        session_id = session_id or str(uuid.uuid4())
        with sqlite3.connect(self.db_path) as conn:
            conn.execute(
                "INSERT INTO sessions (id, platform, created_at, updated_at) VALUES (?, ?, ?, ?)",
                (session_id, platform, int(time.time()), int(time.time()))
            )
        return session_id
    
    def add_message(self, session_id, role, content):
        """添加消息"""
        with sqlite3.connect(self.db_path) as conn:
            conn.execute(
                "INSERT INTO messages (session_id, role, content, timestamp) VALUES (?, ?, ?, ?)",
                (session_id, role, content, int(time.time()))
            )
            conn.execute(
                "INSERT INTO messages_fts (rowid, content) VALUES (last_insert_rowid(), ?)",
                (content,)
            )
    
    def search(self, query, session_id=None):
        """FTS5 搜索"""
        with sqlite3.connect(self.db_path) as conn:
            if session_id:
                cursor = conn.execute(
                    "SELECT m.* FROM messages_fts f JOIN messages m ON f.rowid = m.id "
                    "WHERE f.content MATCH ? AND m.session_id = ?",
                    (query, session_id)
                )
            else:
                cursor = conn.execute(
                    "SELECT m.* FROM messages_fts f JOIN messages m ON f.rowid = m.id "
                    "WHERE f.content MATCH ?",
                    (query,)
                )
            return cursor.fetchall()
```

class CronScheduler:
    def __init__(self):
        self.jobs = []
        self.running = False
    
    def add_job(self, schedule, command, platform=None, chat_id=None):
        """添加定时任务"""
        job = {
            "schedule": schedule,  # cron 表达式
            "command": command,    # 要执行的命令
            "platform": platform,  # 交付平台
            "chat_id": chat_id     # 目标聊天 ID
        }
        self.jobs.append(job)
    
    async def run(self):
        """运行调度器"""
        self.running = True
        while self.running:
            now = datetime.now()
            for job in self.jobs:
                if self._matches_schedule(job["schedule"], now):
                    await self._execute_job(job)
            await asyncio.sleep(60)  # 每分钟检查
    
    async def _execute_job(self, job):
        """执行任务"""

        result = agent.chat(job["command"])
        
        if job["platform"]:
            # 交付到指定平台
